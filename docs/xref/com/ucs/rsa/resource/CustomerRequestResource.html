152 	{
153 
154 		List<CustomerRequestModel> customerRequestModels = new ArrayList<>();
155 		customerRequestModels = getCustomerRequestService().loadAll(CustomerRequestModel.class);
156 
157 		CustomerRequestsDTO customerRequestsDTO = new CustomerRequestsDTO();
158 		List<CustomerRequestDTO> CustomerRequestDTOs = new ArrayList<>();
159 
160 		for (CustomerRequestModel customerRequestModel : customerRequestModels)

162 			CustomerRequestDTO customerRequestDTO = new CustomerRequestDTO();

164 			customerRequestDTO.setCustomerLongitude(customerRequestModel.getCustomerLongitude());
165 			customerRequestDTO.setIssueId(customerRequestModel.getIssueId());
166 			customerRequestDTO.setIssueStatus(customerRequestModel.getIssueStatus());
167 			CustomerRequestDTOs.add(customerRequestDTO);
168 		}
169 		customerRequestsDTO.setCustomerRequests(CustomerRequestDTOs);
170 		return new ModelAndView("xml", "customers", customerRequestsDTO);

251 	/**
252 	 * Send notification.
253 	 *
254 	 * @param entities
255 	 *           the entities
256 	 * @return the string
257 	 */
258 	private String sendNotification(CustomerRequestModel entities)
259 	{
260 		double fixedRange = 0;
261 		double lowStartRating = 0;
262 		double lowEndRating = 0;
263 		double highStartRating = 0;
264 		double highEndRating = 0;
265 		double incrementRange = 0;
266 		int numberOfLoop = 0;
267 		Properties prop = new Properties();
268 		InputStream inputStream = CustomerRequestResource.class.getClassLoader().getResourceAsStream("/constant.properties");
269 		try
270 		{
271 			prop.load(inputStream);
272 			String fixedRangeInString = prop.getProperty("FIXED_RADIUS_RANGE");
273 			String lowStartRatingInString = prop.getProperty("LOW_RATING_START");
274 			String lowEndRatingInString = prop.getProperty("LOW_RATING_END");
275 			String highStartRatingInString = prop.getProperty("HIGH_RATING_START");
276 			String highEndRatingInString = prop.getProperty("HIGH_RATING_END");
277 			String incrementRangeInString = prop.getProperty("INCREASED_RADIUS_RANGE");
278 			String numberOfLoopInString = prop.getProperty("NUMBER_OF_LOOP");
279 			fixedRange = Double.parseDouble(fixedRangeInString);
280 			lowStartRating = Double.parseDouble(lowStartRatingInString);
281 			lowEndRating = Double.parseDouble(lowEndRatingInString);
282 			highStartRating = Double.parseDouble(highStartRatingInString);
283 			highEndRating = Double.parseDouble(highEndRatingInString);
284 			incrementRange = Double.parseDouble(incrementRangeInString);
285 			numberOfLoop = Integer.parseInt(numberOfLoopInString);
286 		}
287 		catch (IOException e1)
288 		{
289 			e1.printStackTrace();
290 		}
291 		String notificationStatus = null;
292 		for (int i = 0; i < numberOfLoop; i++)
293 		{
294 			if (i != 0)
295 			{
296 				fixedRange = fixedRange + incrementRange;
297 			}
298 			ArrayList<Double> ratingArray = new ArrayList<Double>();
299 			ratingArray.add(highStartRating);
300 			ratingArray.add(highEndRating);
301 			ArrayList<Integer> serviceProviderID = new ArrayList<Integer>();
302 			serviceProviderID = getServiceProviderList(entities, fixedRange, ratingArray);
303 			if (serviceProviderID.isEmpty())
304 			{
305 				ratingArray.clear();
306 				ratingArray.add(lowStartRating);
307 				ratingArray.add(lowEndRating);
308 				serviceProviderID = getServiceProviderList(entities, fixedRange, ratingArray);
309 				if (serviceProviderID.isEmpty())
310 				{
311 
312 				}
313 				else
314 				{
315 					notificationStatus = sendGCMNotificationAndGetStatus(serviceProviderID, entities);
316 					break;
317 				}
318 			}
319 			else
320 			{
321 				notificationStatus = sendGCMNotificationAndGetStatus(serviceProviderID, entities);
322 				break;
323 			}
324 		}
325 		return notificationStatus;
326 	}

367 			ArrayList<Double> ratingArray)
368 	{
369 		ArrayList<Integer> serviceProviderID = new ArrayList<Integer>();
370 		String issueType = getServiceTypeWithTypeID(entities.getServiceTypeModel().getServiceTypeId());

373 		double latStoped = entities.getCustomerLatitude() + fixedRange;
