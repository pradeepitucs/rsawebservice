69  	@Override
70  	public ArrayList<String> updateCustomerRequestByEmployee(CustomerRequestModel iCustomerRequestModel)
71  	{
72  		ArrayList<String> customerRequestResponse = new ArrayList<String>();
73  		Session theSession = null;
74  		try
75  		{
76  			theSession = currentSession();
77  			CustomerRequestModel customerRequestList = (CustomerRequestModel) theSession
78  					.createCriteria(CustomerRequestModel.class, "customerRequestModel")
79  					.add(Restrictions.eq("issueId", iCustomerRequestModel.getIssueId())).uniqueResult();
80  			//CustomerRequestModel customerRequestList = cr.list();
81  			int customer_id = customerRequestList.getCustomerModel().getUserId();
82  			customerRequestResponse.add(String.valueOf(customerRequestList.getCustomerModel().getUserId()));
83  			if (customerRequestList.getIssueStatus() == null || customerRequestList.getIssueStatus().isEmpty())
84  			{
85  				customerRequestList.setIssueStartTime(iCustomerRequestModel.getIssueStartTime());
86  				customerRequestList.setIssueStatus(iCustomerRequestModel.getIssueStatus());
87  				customerRequestList.setEmployeeModel(iCustomerRequestModel.getEmployeeModel());
88  				theSession.saveOrUpdate(customerRequestList);
89  				customerRequestResponse.add("Accepted");
90  			}
91  			else
92  			{
93  				customerRequestResponse.add("Accepted by other Person");
94  			}
95  
96  		}
97  		catch (RSAException e)
98  		{
99  			throw e;
100 		}
101 		catch (RuntimeException ex)
102 		{
103 			RSAException rsaEx = new RSAException();
104 			rsaEx.setRootCause(ex);
105 			rsaEx.setError(RSAErrorConstants.ErrorCode.SYSTEM_ERROR);
106 			throw rsaEx;
107 		}
108 		return customerRequestResponse;
109 	}

115 	 */
116 	@SuppressWarnings("null")
117 	@Override
118 	public ArrayList<Integer> getServiceProviderIDS(ArrayList<Double> ratingAndLocation, String serviceType)
119 	{
120 		Session theSession = null;
121 		List<Integer> serviceProviderIDList = new ArrayList<Integer>();
122 		try
123 		{
124 			theSession = currentSession();
125 
126 			Criteria cr = (Criteria) theSession.createCriteria(ServiceProviderModel.class, "customerModel")
127 					.add(Restrictions.between("serviceProviderLatitude", ratingAndLocation.get(2), ratingAndLocation.get(3)))
128 					.add(Restrictions.between("serviceProviderLongitude", ratingAndLocation.get(4), ratingAndLocation.get(5)))
129 					.add(Restrictions.between("rating", ratingAndLocation.get(0), ratingAndLocation.get(1)))
130 					.add(Restrictions.like("serviceproviderExperties", "%" + serviceType + "%"))
131 					.add(Restrictions.like("serviceProviderotherServices", "%" + serviceType + "%"))
132 					.setProjection(Projections.projectionList().add(Projections.property("serviceProviderId"), "serviceProviderId"))
133 					.setResultTransformer(Transformers.aliasToBean(ServiceProviderModel.class));
134 			@SuppressWarnings("unchecked")
135 			List<ServiceProviderModel> serviceProviderList = cr.list();
136 			for (int i = 0; i < serviceProviderList.size(); i++)
137 			{
138 				serviceProviderIDList.add(serviceProviderList.get(i).getServiceProviderId());
139 			}
140 			System.out.println(serviceProviderList);
141 		}
142 		catch (RSAException e)
143 		{
144 			throw e;
145 		}
146 		catch (RuntimeException ex)
147 		{
148 			RSAException rsaEx = new RSAException();
149 			rsaEx.setRootCause(ex);
150 			rsaEx.setError(RSAErrorConstants.ErrorCode.SYSTEM_ERROR);
151 			throw rsaEx;
152 		}
153 		return (ArrayList<Integer>) serviceProviderIDList;
154 	}
