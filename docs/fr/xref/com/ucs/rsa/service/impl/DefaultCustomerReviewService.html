4   package com.ucs.rsa.service.impl;
5   
6   import org.springframework.beans.factory.annotation.Autowired;
7   import org.springframework.beans.factory.annotation.Qualifier;
8   import org.springframework.stereotype.Service;
9   
10  import com.ucs.rsa.common.constants.RSAErrorConstants;
11  import com.ucs.rsa.common.exception.RSAException;
12  import com.ucs.rsa.daos.CustomerReviewDAO;
13  import com.ucs.rsa.model.CustomerReviewModel;
14  import com.ucs.rsa.service.CustomerReviewService;
15  
16  
17  /**
18   * @author Gururaj A M
19   * @version 1.0
20   * 
21   *          The Class DefaultCustomerReviewService.
22   */
23  @Service
24  public class DefaultCustomerReviewService extends DefaultBaseService implements CustomerReviewService
25  {
26  
27  	/** The customer review DAO. */
28  	@Autowired
29  	@Qualifier("customerReviewDAO")
30  	CustomerReviewDAO customerReviewDAO;
31  
32  	/*
33  	 * (non-Javadoc)
34  	 * 
35  	 * @see com.ucs.rsa.service.CustomerReviewService#updateCustomerReview(com.ucs.rsa.model.CustomerReviewModel)
36  	 */
37  	@Override
38  	public CustomerReviewModel updateCustomerReview(CustomerReviewModel iCustomerReviewModel)
39  	{
40  		CustomerReviewModel customerReviewModel = null;
41  		if (!"".equals(iCustomerReviewModel.getCustomerModel()) && !"".equals(iCustomerReviewModel.getEmployeeModel())
42  				&& !"".equals(iCustomerReviewModel.getRating()) && !"".equals(iCustomerReviewModel.getEmployeeModel().getUserId())
43  				&& !"".equals(iCustomerReviewModel.getCustomerModel().getUserId()))
44  		{
45  			customerReviewModel = customerReviewDAO.updateCustomerReview(iCustomerReviewModel);
46  		}
47  		else
48  		{
49  			RSAException rsaEx = new RSAException();
50  			rsaEx.setError(RSAErrorConstants.ErrorCode.MANDATORY_PARAMS_MISSING_ERROR);
51  			throw rsaEx;
52  		}
53  		return customerReviewModel;
54  	}
55  
56  }
