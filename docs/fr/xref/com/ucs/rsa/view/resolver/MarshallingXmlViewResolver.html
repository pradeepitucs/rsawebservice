4   package com.ucs.rsa.view.resolver;

6   
7   
8   import java.util.Locale;
9   
10  import org.springframework.beans.factory.annotation.Autowired;
11  import org.springframework.oxm.Marshaller;
12  import org.springframework.web.servlet.View;
13  import org.springframework.web.servlet.ViewResolver;
14  import org.springframework.web.servlet.view.xml.MarshallingView;
15  
16  
17  /**
18   * @author Gururaj A M
19   * @version 1.0
20   * 
21   *          The Class MarshallingXmlViewResolver.
22   */
23  public class MarshallingXmlViewResolver implements ViewResolver
24  {
25  
26  	/** The marshaller. */
27  	private Marshaller marshaller;
28  
29  	/**
30  	 * Instantiates a new marshalling xml view resolver.
31  	 *
32  	 * @param marshaller
33  	 *           the marshaller
34  	 */
35  	@Autowired
36  	public MarshallingXmlViewResolver(Marshaller marshaller)
37  	{
38  		this.marshaller = marshaller;
39  	}
40  
41  	/**
42  	 * Get the view to use.
43  	 *
44  	 * @param viewName
45  	 *           the view name
46  	 * @param locale
47  	 *           the locale
48  	 * @return the view
49  	 * @throws Exception
50  	 *            the exception
51  	 */
52  	@Override
53  	public View resolveViewName(String viewName, Locale locale) throws Exception
54  	{
55  		MarshallingView view = new MarshallingView();
56  		view.setMarshaller(marshaller);
57  		return view;
58  	}
59  }
